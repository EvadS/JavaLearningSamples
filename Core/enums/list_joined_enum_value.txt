public enum PlatformType {
    WEB(1, "web"),
    ANDROID(2, "Android"),
    IOS(3, "ios");

    private int id;
    private String value;

    PlatformType(int id, String value) {
        this.id = id;
        this.value = value;
    }

    public int getId() {
        return id;
    }

    public String getValue() {
        return value;
    }

    public static PlatformType of(int id) {
        return Stream.of(PlatformType.values())
                .filter(p -> p.getId() == id)
                .findFirst()
                .orElseThrow(IllegalArgumentException::new);
    }
}
